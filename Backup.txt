    <script>
        // Standardwerte aus deinem Profil
        const defaultValues = {
            km_gesamt: 12500,
            km_urlaub: 2000,
            verbrauch_benzin: 6.8,
            preis_benzin: 1.80,
            verbrauch_strom: 18.5,
            preis_strom_arbeit: 0.375,
            preis_strom_urlaub: 0.59,
            steuer_benzin: 140,
            vers_benzin: 650,
            wartung_benzin: 450,
            steuer_elektro: 0,
            vers_elektro: 720,
            wartung_elektro: 250,
            thg_praemie: 80
        };

        let costChart = null;

        // Theme Toggle Logik
        function toggleTheme() {
            const body = document.documentElement;
            const currentTheme = body.getAttribute('data-theme');
            const newTheme = currentTheme === 'dark' ? 'light' : 'dark';
            body.setAttribute('data-theme', newTheme);
            localStorage.setItem('theme', newTheme);
            updateChartTheme(); // Diagrammfarben anpassen
        }

        // Initialisierung beim Laden
        window.onload = () => {
            // Theme laden
            const savedTheme = localStorage.getItem('theme');
            if (savedTheme) {
                document.documentElement.setAttribute('data-theme', savedTheme);
            } else if (window.matchMedia && window.matchMedia('(prefers-color-scheme: dark)').matches) {
                document.documentElement.setAttribute('data-theme', 'dark');
            }

            // Daten laden und Felder füllen
            loadData();
            initChart();
            calculate();
        };

        function loadData() {
            const inputs = document.querySelectorAll('input[data-key]');
            inputs.forEach(input => {
                const key = input.getAttribute('data-key');
                const savedValue = localStorage.getItem(key);
                if (savedValue !== null) {
                    input.value = savedValue;
                } else {
                    input.value = defaultValues[key];
                }
            });
        }

        function resetDefaults() {
            const inputs = document.querySelectorAll('input[data-key]');
            inputs.forEach(input => {
                const key = input.getAttribute('data-key');
                input.value = defaultValues[key];
                localStorage.setItem(key, defaultValues[key]);
            });
            calculate();
        }

        function getVal(id) {
            return parseFloat(document.getElementById(id).value) || 0;
        }

        function formatCurrency(value) {
            return value.toLocaleString('de-DE', { style: 'currency', currency: 'EUR' });
        }

        function calculate() {
            // Eingaben speichern
            const inputs = document.querySelectorAll('input[data-key]');
            inputs.forEach(input => {
                localStorage.setItem(input.getAttribute('data-key'), input.value);
            });

            // Werte auslesen
            const kmGesamt = getVal('km_gesamt');
            const kmUrlaub = getVal('km_urlaub');
            const kmAlltag = Math.max(0, kmGesamt - kmUrlaub);

            // Berechnung Benzin
            const energieBenzin = (kmGesamt / 100) * getVal('verbrauch_benzin') * getVal('preis_benzin');
            const steuerBenzin = getVal('steuer_benzin');
            const versBenzin = getVal('vers_benzin');
            const wartungBenzin = getVal('wartung_benzin');
            const totalBenzin = energieBenzin + steuerBenzin + versBenzin + wartungBenzin;

            // Berechnung Elektro
            const energieStromAlltag = (kmAlltag / 100) * getVal('verbrauch_strom') * getVal('preis_strom_arbeit');
            const energieStromUrlaub = (kmUrlaub / 100) * getVal('verbrauch_strom') * getVal('preis_strom_urlaub');
            const energieElektro = energieStromAlltag + energieStromUrlaub;
            const steuerElektro = getVal('steuer_elektro') - getVal('thg_praemie'); // THG als negativer Steuerposten
            const versElektro = getVal('vers_elektro');
            const wartungElektro = getVal('wartung_elektro');
            const totalElektro = energieElektro + steuerElektro + versElektro + wartungElektro;

            // Tabelle aktualisieren
            document.getElementById('res_energie_b').innerText = formatCurrency(energieBenzin);
            document.getElementById('res_energie_e').innerText = formatCurrency(energieElektro);
            document.getElementById('res_steuer_b').innerText = formatCurrency(steuerBenzin);
            document.getElementById('res_steuer_e').innerText = formatCurrency(steuerElektro);
            document.getElementById('res_vers_b').innerText = formatCurrency(versBenzin);
            document.getElementById('res_vers_e').innerText = formatCurrency(versElektro);
            document.getElementById('res_wartung_b').innerText = formatCurrency(wartungBenzin);
            document.getElementById('res_wartung_e').innerText = formatCurrency(wartungElektro);
            
            document.getElementById('res_total_b').innerText = formatCurrency(totalBenzin);
            document.getElementById('res_total_e').innerText = formatCurrency(totalElektro);

            // Chart aktualisieren
            updateChart([energieBenzin, versBenzin, wartungBenzin, steuerBenzin], 
                        [energieElektro, versElektro, wartungElektro, steuerElektro]);

            // Berechne die Differenz
            const diff = totalBenzin - totalElektro;

            // Suche die Stelle, wo die Tabellenwerte gesetzt werden und füge dies hinzu:
            const savingsElement = document.getElementById('res_savings_final');
            savingsElement.innerText = diff.toLocaleString('de-DE', {style:'currency', currency:'EUR'});

            // Optional: Farbe anpassen, falls Elektro teurer wird (Minus-Ersparnis)
            if (diff >= 0) {
                savingsElement.style.color = "var(--accent-elektro)";
            } else {
                savingsElement.style.color = "var(--accent-benzin)";
            }
        }

        function initChart() {
            const ctx = document.getElementById('costChart').getContext('2d');
            
            Chart.defaults.color = getComputedStyle(document.documentElement).getPropertyValue('--text-color');
            Chart.defaults.font.family = '-apple-system, BlinkMacSystemFont, "Segoe UI", Roboto, sans-serif';

            costChart = new Chart(ctx, {
                type: 'bar',
                data: {
                    labels: ['Benzin (Tucson)', 'Elektro (SUV)'],
                    datasets: [
                        { label: 'Energie', data: [0, 0], backgroundColor: '#3498db' },
                        { label: 'Versicherung', data: [0, 0], backgroundColor: '#9b59b6' },
                        { label: 'Wartung', data: [0, 0], backgroundColor: '#f1c40f' },
                        { label: 'Steuer (inkl. THG)', data: [0, 0], backgroundColor: '#e67e22' }
                    ]
                },
                options: {
                    responsive: true,
                    maintainAspectRatio: false,
                    plugins: {
                        legend: { position: 'bottom' },
                        tooltip: {
                            callbacks: {
                                label: function(context) {
                                    let label = context.dataset.label || '';
                                    if (label) { label += ': '; }
                                    if (context.parsed.y !== null) {
                                        label += new Intl.NumberFormat('de-DE', { style: 'currency', currency: 'EUR' }).format(context.parsed.y);
                                    }
                                    return label;
                                }
                            }
                        }
                    },
                    scales: {
                        x: { stacked: true, grid: { display: false } },
                        y: { stacked: true, border: { display: false } }
                    }
                }
            });
        }

        function updateChart(dataBenzin, dataElektro) {
            if(!costChart) return;
            // Energie
            costChart.data.datasets[0].data = [dataBenzin[0], Math.max(0, dataElektro[0])];
            // Versicherung
            costChart.data.datasets[1].data = [dataBenzin[1], Math.max(0, dataElektro[1])];
            // Wartung
            costChart.data.datasets[2].data = [dataBenzin[2], Math.max(0, dataElektro[2])];
            // Steuer/THG
            costChart.data.datasets[3].data = [dataBenzin[3], dataElektro[3]];
            
            costChart.update();
        }

        function updateChartTheme() {
            if(costChart) {
                const textColor = getComputedStyle(document.documentElement).getPropertyValue('--text-color').trim();
                const gridColor = getComputedStyle(document.documentElement).getPropertyValue('--border-color').trim();
                
                costChart.options.plugins.legend.labels.color = textColor;
                costChart.options.scales.x.ticks.color = textColor;
                costChart.options.scales.y.ticks.color = textColor;
                costChart.options.scales.y.grid.color = gridColor;
                costChart.update();
            }
        }
